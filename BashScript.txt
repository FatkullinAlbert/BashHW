#!/usr/bin/env bash

# ==================================================
# Скрипт ежечасного отчёта о работе веб-сервера
# Предназначен для запуска из cron
# ==================================================

# ------------------ Конфигурация ------------------
LOG_FILE="/var/log/nginx/access.log"            # путь к логу доступа
EMAIL="$USER"		                        # email для отправки отчёта
LOCK_FILE="/tmp/web_report.lock"                # файл блокировки
LAST_RUN_FILE="/tmp/web_report_lastrun"         # файл с временем последнего запуска (Unix time)
# --------------------------------------------------

# Функция для преобразования даты из лога в Unix time
# Формат строки: [10/Oct/2023:13:55:36 +0000]
logdate2epoch() {
    local logdate="$1"
    # Удаляем квадратные скобки и разделяем на дату и зону
    logdate=${logdate#[}   # убрать открывающую скобку
    logdate=${logdate%]}   # убрать закрывающую скобку
    # logdate теперь имеет вид: 10/Oct/2023:13:55:36 +0000
    local date_part="${logdate% *}"      # 10/Oct/2023:13:55:36
    local zone_part="${logdate#* }"      # +0000
    # Заменяем '/' и ':' на пробелы для date -d
    local d=${date_part//\// }           # 10 Oct 2023:13:55:36
    d=${d//:/ }                           # 10 Oct 2023 13 55 36
    # Собираем строку: "10 Oct 2023 13:55:36 +0000"
    local date_str="${d:0:11} ${d:11:2}:${d:14:2}:${d:17:2} $zone_part"
    date -d "$date_str" +%s 2>/dev/null
}

# Предотвращение одновременного запуска (flock)
exec 200>"$LOCK_FILE"
flock -n 200 || { echo "Скрипт уже выполняется, выход"; exit 1; }

# Определяем временной диапазон
current_epoch=$(date +%s)
if [[ -f "$LAST_RUN_FILE" ]]; then
    last_epoch=$(cat "$LAST_RUN_FILE")
else
    # Первый запуск – берём последний час
    last_epoch=$((current_epoch - 3600))
fi

# Инициализация ассоциативных массивов для сбора статистики
declare -A ip_count
declare -A url_count
declare -A http_code_count
errors=()   # массив для строк с ошибками (4xx, 5xx)

# Обработка лога: идём с конца файла до достижения записей старше last_epoch
while IFS= read -r line; do
    # Извлекаем дату из строки (поле в квадратных скобках)
    if [[ $line =~ \[([^]]+)\] ]]; then
        logdate="${BASH_REMATCH[1]}"
        line_epoch=$(logdate2epoch "$logdate")
        # Если не удалось распарсить дату, пропускаем строку
        [[ -z "$line_epoch" ]] && continue
        # Достигли записей старше last_epoch – прекращаем обработку
        if (( line_epoch < last_epoch )); then
            break
        fi
    else
        # Нет даты – пропускаем (на всякий случай)
        continue
    fi

    # Извлекаем IP (первое поле до пробела)
    ip=$(echo "$line" | awk '{print $1}')
    # Извлекаем URL (типичный combined: "GET /path HTTP/1.1" в кавычках)
    # Берём содержимое между двойными кавычками после метода
    if [[ $line =~ \".*[[:space:]]([^[:space:]]+)[[:space:]]HTTP/ ]]; then
        url="${BASH_REMATCH[1]}"
    else
        url="unknown"
    fi
    # Извлекаем HTTP-код ответа (предпоследнее поле)
    http_code=$(echo "$line" | awk '{print $(NF-1)}')
    # Проверяем, что код состоит из трёх цифр (иначе пропускаем)
    if [[ ! $http_code =~ ^[0-9]{3}$ ]]; then
        http_code="unknown"
    fi

    # Счётчики
    ((ip_count["$ip"]++))
    ((url_count["$url"]++))
    ((http_code_count["$http_code"]++))

    # Если код 4xx или 5xx – сохраняем строку как ошибку
    if [[ $http_code =~ ^[45][0-9]{2}$ ]]; then
        errors+=("$line")
    fi
done < <(tac "$LOG_FILE" 2>/dev/null)

# Формирование отчёта
report=$(mktemp)
{
    echo "Отчёт о работе веб-сервера"
    echo "Временной диапазон: $(date -d @$last_epoch) - $(date -d @$current_epoch)"
    echo "=================================================="
    echo
    echo "IP-адреса с наибольшим числом запросов:"
    if [[ ${#ip_count[@]} -gt 0 ]]; then
        # Сортируем по убыванию и выводим топ-10
        for ip in "${!ip_count[@]}"; do
            echo "$ip ${ip_count[$ip]}"
        done | sort -rn -k2 | head -10 | while read ip cnt; do
            printf "  %-15s %d\n" "$ip" "$cnt"
        done
    else
        echo "  Нет данных"
    fi
    echo

    echo "Запрашиваемые URL с наибольшим числом запросов:"
    if [[ ${#url_count[@]} -gt 0 ]]; then
        for url in "${!url_count[@]}"; do
            echo "$url ${url_count[$url]}"
        done | sort -rn -k2 | head -10 | while read url cnt; do
            printf "  %-30s %d\n" "${url:0:30}" "$cnt"
        done
    else
        echo "  Нет данных"
    fi
    echo

    echo "HTTP-коды ответов:"
    if [[ ${#http_code_count[@]} -gt 0 ]]; then
        for code in "${!http_code_count[@]}"; do
            echo "  $code: ${http_code_count[$code]}"
        done | sort
    else
        echo "  Нет данных"
    fi
    echo

    echo "Ошибки веб-сервера/приложения (коды 4xx и 5xx):"
    if [[ ${#errors[@]} -gt 0 ]]; then
        # Показываем последние 20 ошибок (чтобы не переполнить письмо)
        for ((i=0; i<${#errors[@]} && i<20; i++)); do
            echo "  ${errors[$i]}"
        done
        if [[ ${#errors[@]} -gt 20 ]]; then
            echo "  ... и ещё $((${#errors[@]}-20)) записей"
        fi
    else
        echo "  Ошибок не обнаружено"
    fi
} > "$report"

# Отправка письма
mail -s "Web Server Report $(date '+%Y-%m-%d %H:%M')" "$EMAIL" < "$report"

# Удаляем временный файл
rm -f "$report"

# Обновляем время последнего запуска
echo "$current_epoch" > "$LAST_RUN_FILE"

exit 0
